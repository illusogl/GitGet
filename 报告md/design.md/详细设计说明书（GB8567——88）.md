### 3.6 算法

#### 指数退避重试算法

```csharp
async Task<T> RetryWithBackoffAsync<T>(Func<Task<T>> requestFn, int maxRetries = 3)
{
    for (int attempt = 1; attempt <= maxRetries; attempt++)
    {
        try
        {
            return await requestFn();
        }
        catch (Exception ex) when (attempt < maxRetries)
        {
            if (ex is RateLimitExceededException rateEx)
            {
                var waitMs = (rateEx.ResetTime - DateTime.UtcNow).TotalMilliseconds + 1000;
                await Task.Delay((int)Math.Min(waitMs, 30000));
            }
            else
            {
                var waitMs = Math.Min(Math.Pow(2, attempt) * 1000, 10000);
                await Task.Delay((int)waitMs);
            }
        }
    }
    throw new InvalidOperationException("Max retries exceeded");
}
```

#### 速率限制管理算法

```csharp
void HandleRateLimit(HttpResponseMessage response)
{
    if (!response.Headers.TryGetValues("X-RateLimit-Remaining", out var remainingValues))
        return;
    if (!response.Headers.TryGetValues("X-RateLimit-Reset", out var resetValues))
        return;

    var remaining = int.Parse(remainingValues.First());
    var resetUnix = long.Parse(resetValues.First());
    var resetTime = DateTimeOffset.FromUnixTimeSeconds(resetUnix);

    if (remaining < 10)
    {
        _logger.Warning("Rate limit low: {Remaining}, resets at {ResetTime}", remaining, resetTime);
    }

    if (remaining == 0)
    {
        throw new RateLimitExceededException(resetTime);
    }
}
```

#### C# 下载实现（核心逻辑）

```csharp
public async Task StartTaskAsync(string taskId, CancellationToken ct)
{
    if (!_tasks.TryGetValue(taskId, out var task))
        throw new KeyNotFoundException($"Task {taskId} not found");

    task.Status = DownloadStatus.Downloading;
    var tempFile = task.TargetPath + ".gitget.part";

    try
    {
        using var httpClient = _httpClientFactory.CreateClient("DownloadClient");
        var request = new HttpRequestMessage(HttpMethod.Get, task.AssetUrl);

        // 断点续传：从已下载的位置开始
        if (File.Exists(tempFile))
        {
            var existingLength = new FileInfo(tempFile).Length;
            request.Headers.Range = new RangeHeaderValue(existingLength, null);
        }

        using var response = await httpClient.SendAsync(request, HttpCompletionOption.ResponseHeadersRead, ct);
        response.EnsureSuccessStatusCode();

        var totalBytes = task.TotalBytes > 0 ? task.TotalBytes : response.Content.Headers.ContentLength ?? 0;
        using var stream = await response.Content.ReadAsStreamAsync(ct);
        using var fileStream = new FileStream(tempFile, FileMode.Append, FileAccess.Write, FileShare.None, 81920, true);

        var buffer = new byte[81920];
        long receivedBytes = new FileInfo(tempFile).Length;
        var startTime = DateTime.UtcNow;

        int bytesRead;
        while ((bytesRead = await stream.ReadAsync(buffer, ct)) > 0)
        {
            await fileStream.WriteAsync(buffer.AsMemory(0, bytesRead), ct);
            receivedBytes += bytesRead;

            // 计算进度
            var elapsed = (DateTime.UtcNow - startTime).TotalSeconds;
            var progress = new DownloadProgress(
                Percent: totalBytes > 0 ? (double)receivedBytes / totalBytes * 100 : 0,
                SpeedBytesPerSec: elapsed > 0 ? receivedBytes / elapsed : 0,
                Eta: totalBytes > 0 && elapsed > 0
                    ? TimeSpan.FromSeconds((totalBytes - receivedBytes) / (receivedBytes / elapsed))
                    : TimeSpan.MaxValue,
                ReceivedBytes: receivedBytes,
                TotalBytes: totalBytes
            );

            _progressReporter?.Report(progress);
            await SaveTaskStateAsync(task);
        }

        // 下载完成：重命名临时文件
        File.Move(tempFile, task.TargetPath, overwrite: true);
        task.Status = DownloadStatus.Completed;
        await SaveTaskStateAsync(task);
    }
    catch (OperationCanceledException)
    {
        task.Status = DownloadStatus.Paused;  // 用户暂停
        await SaveTaskStateAsync(task);
    }
    catch (Exception ex)
    {
        task.Status = DownloadStatus.Failed;
        task.ErrorMessage = ex.Message;
        await SaveTaskStateAsync(task);
    }
}
```

### 5.5 算法

#### 兴趣模型构建算法

```csharp
public InterestModel BuildInterestModel(IReadOnlyList<Repo> stars)
{
    var languageWeights = new Dictionary<string, double>();
    var topicWeights = new Dictionary<string, double>();

    foreach (var repo in stars)
    {
        // 语言权重
        if (!string.IsNullOrEmpty(repo.Language))
        {
            var weight = 1.0 + (repo.StargazersCount / 10000.0);  // 大项目加权
            languageWeights[repo.Language] = languageWeights.GetValueOrDefault(repo.Language) + weight;
        }

        // 话题标签权重
        foreach (var topic in repo.Topics)
        {
            topicWeights[topic] = topicWeights.GetValueOrDefault(topic) + 1;
        }
    }

    // 归一化后取 Top 5 语言和 Top 10 话题
    var topLanguages = NormalizeAndTakeTop(languageWeights, 5);
    var topTopics = NormalizeAndTakeTop(topicWeights, 10);

    return new InterestModel(topLanguages, topTopics);
}
```

#### 推荐查询算法

```csharp
public async Task<IReadOnlyList<Repo>> GetPersonalizedAsync(string token, CancellationToken ct = default)
{
    // 1. 获取用户信息
    var user = await _gitHubApi.GetUserInfoAsync(ct);
    
    // 2. 获取 Star 列表
    var stars = await _gitHubApi.GetUserStarsAsync(user.Login, ct: ct);
    
    // 3. 构建兴趣模型
    var model = BuildInterestModel(stars);
    
    // 4. 按语言权重并行查询推荐
    var queryTasks = model.TopLanguages.Keys
        .Select(lang => _gitHubApi.SearchReposAsync(
            new SearchParams(Keyword: $"language:{lang}", Sort: "stars", PerPage: 5),
            ct))
        .ToArray();
    
    var results = await Task.WhenAll(queryTasks);
    
    // 5. 合并、去重、过滤已 Star 的
    var allRepos = results
        .SelectMany(r => r.Items)
        .DistinctBy(r => r.FullName)
        .Where(r => !stars.Any(s => s.FullName == r.FullName))
        .ToList();
    
    // 6. 按综合权重排序
    allRepos.Sort((a, b) => CombinedScore(b, model).CompareTo(CombinedScore(a, model)));
    
    return allRepos.Take(20).ToList();
}
```

#### 匹配算法

```csharp
public MatchResult MatchAsset(IReadOnlyList<Asset> assets, PlatformInfo? platform = null)
{
    platform ??= GetPlatformInfo();
    
    var scored = new List<(Asset Asset, int Score)>();

    foreach (var asset in assets)
    {
        var maxScore = 0;
        
        if (MatchRules.TryGetValue(platform.OS, out var rules))
        {
            foreach (var rule in rules)
            {
                if (rule.Pattern.IsMatch(asset.Name))
                {
                    maxScore = Math.Max(maxScore, rule.Priority);
                }
            }
        }
        
        if (maxScore > 0)
        {
            scored.Add((asset, maxScore));
        }
    }

    scored.Sort((a, b) => b.Score.CompareTo(a.Score));

    return new MatchResult(
        Matched: scored.Select(s => s.Asset).ToList(),
        Recommended: scored.FirstOrDefault().Asset
    );
}
```

```

    ↓
for each asset in assets:
    ↓
[文件名正则匹配] → [win规则: .exe, .msi, -win*等]
    ↓
[取最高优先级得分]
    ↓
[按得分降序排序]
    ↓
return { Matched: [...], Recommended: highest }
```

```csharp
public async Task<TokenInfo> HandleCallbackAsync(string code, CancellationToken ct = default)
{
    // 1. 构建 OAuth Token 请求
    var request = new HttpRequestMessage(HttpMethod.Post, "https://github.com/login/oauth/access_token");
    request.Content = new FormUrlEncodedContent(new Dictionary<string, string>
    {
        ["client_id"] = _clientId,
        ["client_secret"] = _clientSecret,
        ["code"] = code,
        ["redirect_uri"] = _redirectUri
    });
    request.Headers.Accept.Add(new MediaTypeWithQualityHeaderValue("application/json"));

    // 2. 发送请求获取 Token
    using var response = await _httpClient.SendAsync(request, ct);
    response.EnsureSuccessStatusCode();
    var tokenResponse = await response.Content.ReadFromJsonAsync<GitHubTokenResponse>(ct);

    // 3. 构建 TokenInfo
    var tokenInfo = new TokenInfo(
        AccessToken: tokenResponse!.AccessToken,
        TokenType: tokenResponse.TokenType,
        Scope: tokenResponse.Scope,
        ExpiresIn: tokenResponse.ExpiresIn,
        RefreshToken: tokenResponse.RefreshToken,
        ExpiresAt: DateTime.UtcNow.AddSeconds(tokenResponse.ExpiresIn)
    );

    // 4. 加密存储到安全存储
    await _secureStore.SaveAsync("github_token", tokenInfo, ct);

    return tokenInfo;
}
```
